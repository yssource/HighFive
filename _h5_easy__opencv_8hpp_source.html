29 inline static std::vector<size_t> shape(const T& data) {
30 return std::vector<size_t>{static_cast<size_t>(data.rows), static_cast<size_t>(data.cols)};
31 }
32
33 inline static std::vector<int> shape(const File& file,
34 const std::string& path,
35 std::vector<size_t> dims) {
36 if (dims.size() == 1) {
37 return std::vector<int>{static_cast<int>(dims[0]), 1ul};
38 }
39 if (dims.size() == 2) {
40 return std::vector<int>{static_cast<int>(dims[0]), static_cast<int>(dims[1])};
41 }
42
43 throw detail::error(file, path, "H5Easy::load: Inconsistent rank");
44 }
45
46 inline static DataSet dump(File& file,
47 const std::string& path,
48 const T& data,
49 const DumpOptions& options) {
50 using value_type = typename T::value_type;
51 DataSet dataset = initDataset<value_type>(file, path, shape(data), options);
52 std::vector<value_type> v(data.begin(), data.end());
53 dataset.write_raw(v.data());
54 if (options.flush()) {
55 file.flush();
56 }
57 return dataset;
58 }

60 inline static T load(const File& file, const std::string& path) {
61 using value_type = typename T::value_type;
62 DataSet dataset = file.getDataSet(path);
63 std::vector<int> dims = shape(file, path, dataset.getDimensions());
64 T data(dims[0], dims[1]);
65 dataset.read(reinterpret_cast<value_type*>(data.data));
66 return data;
67 }

69 inline static Attribute dumpAttribute(File& file,
70 const std::string& path,
71 const std::string& key,
72 const T& data,
73 const DumpOptions& options) {
74 using value_type = typename T::value_type;
75 Attribute attribute = initAttribute<value_type>(file, path, key, shape(data), options);
76 std::vector<value_type> v(data.begin(), data.end());
77 attribute.write_raw(v.data());
78 if (options.flush()) {
79 file.flush();
80 }
81 return attribute;
82 }
83
84 inline static T loadAttribute(const File& file,
85 const std::string& path,
86 const std::string& key) {
87 using value_type = typename T::value_type;
88 DataSet dataset = file.getDataSet(path);
89 Attribute attribute = dataset.getAttribute(key);
90 DataSpace dataspace = attribute.getSpace();
91 std::vector<int> dims = shape(file, path, dataspace.getDimensions());
92 T data(dims[0], dims[1]);
93 attribute.read(reinterpret_cast<value_type*>(data.data));
94 return data;
95 }
