83 static RegularHyperSlab fromHDF5Sizes(std::vector<hsize_t> offset_,
84 std::vector<hsize_t> count_ = {},
85 std::vector<hsize_t> stride_ = {},
86 std::vector<hsize_t> block_ = {}) {

90 slab.stride = stride_;
91 slab.block = block_;

95
96 size_t rank() const {
97 return std::max(std::max(offset.size(), count.size()),
98 std::max(stride.size(), block.size()));
99 }
100
102 std::vector<size_t> packedDims() const {
103 auto n_dims = rank();
104 auto dims = std::vector<size_t>(n_dims, 0);
105
106 for (size_t i = 0; i < n_dims; ++i) {
107 dims[i] = count[i] * (block.empty() ? 1 : block[i]);
108 }
109
110 return dims;
111 }
112
113 std::vector<hsize_t> offset;
114 std::vector<hsize_t> count;
115 std::vector<hsize_t> stride;
116 std::vector<hsize_t> block;

172 DataSpace apply(const DataSpace& space_) const {
173 auto space = space_.clone();
174 for (const auto& sel: selects) {
175 if (sel.op == Op::None) {
176 H5Sselect_none(space.getId());
177 } else {
178 auto error_code =
179 H5Sselect_hyperslab(space.getId(),
180 convert(sel.op),
181 sel.offset.empty() ? nullptr : sel.offset.data(),
182 sel.stride.empty() ? nullptr : sel.stride.data(),
183 sel.count.empty() ? nullptr : sel.count.data(),
184 sel.block.empty() ? nullptr : sel.block.data());
185
186 if (error_code < 0) {
187 HDF5ErrMapper::ToException<DataSpaceException>("Unable to select hyperslab");
188 }
189 }
190 }
191 return space;
192 }

209 H5S_seloper_t convert(Op op) const {
210 switch (op) {

212 return H5S_SELECT_NOOP;
213 case Op::Set:
214 return H5S_SELECT_SET;
215 case Op::Or:
216 return H5S_SELECT_OR;
217 case Op::And:
218 return H5S_SELECT_AND;
219 case Op::Xor:
220 return H5S_SELECT_XOR;
221 case Op::NotB:
222 return H5S_SELECT_NOTB;
223 case Op::NotA:
224 return H5S_SELECT_NOTA;
225 case Op::Append:
226 return H5S_SELECT_APPEND;
227 case Op::Prepend:
228 return H5S_SELECT_PREPEND;
229 case Op::Invalid:
230 return H5S_SELECT_INVALID;
231 default:
232 throw DataSpaceException("Invalid HyperSlab operation.");
233 }
