111class CompoundType: public DataType {
112 public:
115 struct member_def {
116 member_def(std::string t_name, DataType t_base_type, size_t t_offset = 0)
117 : name(std::move(t_name))
118 , base_type(std::move(t_base_type))
119 , offset(t_offset) {}
120 std::string name;
121 DataType base_type;
122 size_t offset;
123 };
124
125 CompoundType(const CompoundType& other) = default;
126
131 inline CompoundType(const std::vector<member_def>& t_members, size_t size = 0)
132 : members(t_members) {
133 create(size);
134 }
135 inline CompoundType(std::vector<member_def>&& t_members, size_t size = 0)
136 : members(std::move(t_members)) {
137 create(size);
138 }
139 inline CompoundType(const std::initializer_list<member_def>& t_members, size_t size = 0)
140 : members(t_members) {
141 create(size);
142 }
143
147 inline CompoundType(DataType&& type)
148 : DataType(type) {
149 if (getClass() != DataTypeClass::Compound) {
150 std::ostringstream ss;
151 ss << "hid " << _hid << " does not refer to a compound data type";
152 throw DataTypeException(ss.str());
153 }
154 int result = H5Tget_nmembers(_hid);
155 if (result < 0) {
156 throw DataTypeException("Could not get members of compound datatype");
157 }
158 size_t n_members = static_cast<size_t>(result);
159 members.reserve(n_members);
160 for (unsigned i = 0; i < n_members; i++) {
161 const char* name = H5Tget_member_name(_hid, i);
162 size_t offset = H5Tget_member_offset(_hid, i);
163 hid_t member_hid = H5Tget_member_type(_hid, i);
164 DataType member_type{member_hid};
165 members.emplace_back(name, member_type, offset);
166 }
167 }
168
172 inline void commit(const Object& object, const std::string& name) const;
173
175 inline const std::vector<member_def>& getMembers() const noexcept {
176 return members;
177 }
178
179 private:
181 std::vector<member_def> members;
182
186 void create(size_t size = 0);
187};
