58template <typename T>
59inline void Attribute::read(T& array) const {
60 const DataSpace& mem_space = getMemSpace();
61 const details::BufferInfo<T> buffer_info(getDataType(),
62 [this]() -> std::string { return this->getName(); });
63
64 if (!details::checkDimensions(mem_space, buffer_info.n_dimensions)) {
65 std::ostringstream ss;
66 ss << "Impossible to read DataSet of dimensions " << mem_space.getNumberDimensions()
67 << " into arrays of dimensions " << buffer_info.n_dimensions;
68 throw DataSpaceException(ss.str());
69 }
70 details::data_converter<T> converter(mem_space);
71 read(converter.transform_read(array), buffer_info.data_type);
72 // re-arrange results
73 converter.process_result(array);
74}

76template <typename T>
77inline void Attribute::read(T* array, const DataType& dtype) const {
78 static_assert(!std::is_const<T>::value,
79 "read() requires a non-const structure to read data into");
80 using element_type = typename details::inspector<T>::base_type;
81 // Auto-detect mem datatype if not provided
82 const DataType& mem_datatype = dtype.empty() ? create_and_check_datatype<element_type>()
83 : dtype;
84
85 if (H5Aread(getId(), mem_datatype.getId(), static_cast<void*>(array)) < 0) {
86 HDF5ErrMapper::ToException<AttributeException>("Error during HDF5 Read: ");
87 }
88}

90template <typename T>
91inline void Attribute::write(const T& buffer) {
92 const DataSpace& mem_space = getMemSpace();
93 const details::BufferInfo<T> buffer_info(getDataType(),
94 [this]() -> std::string { return this->getName(); });
95
96 if (!details::checkDimensions(mem_space, buffer_info.n_dimensions)) {
97 std::ostringstream ss;
98 ss << "Impossible to write buffer of dimensions " << buffer_info.n_dimensions
99 << " into dataset of dimensions " << mem_space.getNumberDimensions();
100 throw DataSpaceException(ss.str());
101 }
102 details::data_converter<T> converter(mem_space);
103 write_raw(converter.transform_write(buffer), buffer_info.data_type);
104}

112 HDF5ErrMapper::ToException<DataSetException>("Error during HDF5 Write: ");
