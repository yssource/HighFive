34template <typename Derivate>
35inline DataSet NodeTraits<Derivate>::createDataSet(const std::string& dataset_name,
36 const DataSpace& space,
37 const DataType& dtype,
38 const DataSetCreateProps& createProps,
39 const DataSetAccessProps& accessProps,
40 bool parents) {
41 LinkCreateProps lcpl;
42 lcpl.add(CreateIntermediateGroup(parents));
43 const auto hid = H5Dcreate2(static_cast<Derivate*>(this)->getId(),
44 dataset_name.c_str(),
45 dtype._hid,
46 space._hid,
47 lcpl.getId(),
48 createProps.getId(),
49 accessProps.getId());
50 if (hid < 0) {

52 std::string("Unable to create the dataset \"") + dataset_name + "\":");

182inline bool NodeTraits<Derivate>::rename(const std::string& src_path,
183 const std::string& dst_path,
184 bool parents) const {
185 LinkCreateProps lcpl;
186 lcpl.add(CreateIntermediateGroup(parents));
187 herr_t status = H5Lmove(static_cast<const Derivate*>(this)->getId(),
188 src_path.c_str(),
189 static_cast<const Derivate*>(this)->getId(),
190 dst_path.c_str(),
191 lcpl.getId(),
192 H5P_DEFAULT);
193 if (status < 0) {
194 HDF5ErrMapper::ToException<GroupException>(std::string("Unable to move link to \"") +
195 dst_path + "\":");
196 return false;
197 }
198 return true;

201template <typename Derivate>
202inline std::vector<std::string> NodeTraits<Derivate>::listObjectNames() const {
203 std::vector<std::string> names;
204 details::HighFiveIterateData iterateData(names);
205
206 size_t num_objs = getNumberObjects();
207 names.reserve(num_objs);
208
209 if (H5Literate(static_cast<const Derivate*>(this)->getId(),
210 H5_INDEX_NAME,
211 H5_ITER_INC,
212 NULL,
213 &details::internal_high_five_iterate<H5L_info_t>,
214 static_cast<void*>(&iterateData)) < 0) {
215 HDF5ErrMapper::ToException<GroupException>(std::string("Unable to list objects in group"));
216 }
217
218 return names;
219}

221template <typename Derivate>
222inline bool NodeTraits<Derivate>::_exist(const std::string& node_name, bool raise_errors) const {
223 SilenceHDF5 silencer{};
224 const auto val =
225 H5Lexists(static_cast<const Derivate*>(this)->getId(), node_name.c_str(), H5P_DEFAULT);
226 if (val < 0) {
227 if (raise_errors) {
228 HDF5ErrMapper::ToException<GroupException>("Invalid link for exist()");
229 } else {
230 return false;
231 }
232 }
233
234 // The root path always exists, but H5Lexists return 0 or 1
235 // depending of the version of HDF5, so always return true for it
236 // We had to call H5Lexists anyway to check that there are no errors
237 return (node_name == "/") ? true : (val > 0);
238}

240template <typename Derivate>
241inline bool NodeTraits<Derivate>::exist(const std::string& group_path) const {
242 // When there are slashes, first check everything is fine
243 // so that subsequent errors are only due to missing intermediate groups
244 if (group_path.find('/') != std::string::npos) {
245 _exist("/"); // Shall not throw under normal circumstances
246 // Unless "/" (already checked), verify path exists (not throwing errors)
247 return (group_path == "/") ? true : _exist(group_path, false);
248 }
249 return _exist(group_path);
250}

263// convert internal link types to enum class.
264// This function is internal, so H5L_TYPE_ERROR shall be handled in the calling context
265static inline LinkType _convert_link_type(const H5L_type_t& ltype) noexcept {
266 switch (ltype) {
267 case H5L_TYPE_HARD:
268 return LinkType::Hard;
269 case H5L_TYPE_SOFT:
270 return LinkType::Soft;
271 case H5L_TYPE_EXTERNAL:
272 return LinkType::External;
273 default:
274 // Other link types are possible but are considered strange to HighFive.
275 // see https://support.hdfgroup.org/HDF5/doc/RM/H5L/H5Lregister.htm
276 return LinkType::Other;
277 }
278}

280template <typename Derivate>
281inline LinkType NodeTraits<Derivate>::getLinkType(const std::string& node_name) const {
282 H5L_info_t linkinfo;
283 if (H5Lget_info(static_cast<const Derivate*>(this)->getId(),
284 node_name.c_str(),
285 &linkinfo,
286 H5P_DEFAULT) < 0 ||
287 linkinfo.type == H5L_TYPE_ERROR) {
288 HDF5ErrMapper::ToException<GroupException>(std::string("Unable to obtain info for link ") +
289 node_name);
290 }
291 return _convert_link_type(linkinfo.type);
