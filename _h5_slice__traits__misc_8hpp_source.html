74template <typename Derivate>
75inline Selection SliceTraits<Derivate>::select_impl(const HyperSlab& hyperslab,
76 const DataSpace& memspace) const {
77 // Note: The current limitation are that memspace must describe a
78 // packed memspace.
79 //
80 // The reason for this is that we're unable to unpack general
81 // hyperslabs when the memory is not contiguous, e.g.
82 // `std::vector<std::vector<double>>`.
83 const auto& slice = static_cast<const Derivate&>(*this);
84 auto filespace = hyperslab.apply(slice.getSpace());
85
86 return Selection(memspace, filespace, details::get_dataset(slice));
87}

89template <typename Derivate>
90inline Selection SliceTraits<Derivate>::select(const HyperSlab& hyper_slab) const {
91 const auto& slice = static_cast<const Derivate&>(*this);
92 auto filespace = slice.getSpace();
93 filespace = hyper_slab.apply(filespace);
94
95 auto n_elements = H5Sget_select_npoints(filespace.getId());
96 auto memspace = DataSpace(std::array<size_t, 1>{size_t(n_elements)});
97
98 return Selection(memspace, filespace, details::get_dataset(slice));
99}

112template <typename Derivate>
113inline Selection SliceTraits<Derivate>::select(const std::vector<size_t>& columns) const {
114 const auto& slice = static_cast<const Derivate&>(*this);
115 const DataSpace& space = slice.getSpace();
116 std::vector<size_t> dims = space.getDimensions();
117
118 std::vector<size_t> counts = dims;
119 counts.back() = 1;
120
121 std::vector<size_t> offsets(dims.size(), 0);
122
123 HyperSlab slab;
124 for (const auto& column: columns) {
125 offsets.back() = column;
126 slab |= RegularHyperSlab(offsets, counts);
127 }
128
129 std::vector<size_t> memdims = dims;
130 memdims.back() = columns.size();
131
132 return select_impl(slab, DataSpace(memdims));
133}

135template <typename Derivate>
136inline Selection SliceTraits<Derivate>::select(const ElementSet& elements) const {
137 const auto& slice = static_cast<const Derivate&>(*this);
138 const hsize_t* data = nullptr;
139 const DataSpace space = slice.getSpace().clone();
140 const std::size_t length = elements._ids.size();
141 if (length % space.getNumberDimensions() != 0) {
142 throw DataSpaceException(
143 "Number of coordinates in elements picking "
144 "should be a multiple of the dimensions.");
145 }
146 const std::size_t num_elements = length / space.getNumberDimensions();
147 std::vector<hsize_t> raw_elements;
148
149 // optimised at compile time
150 // switch for data conversion on 32bits platforms
151 if (std::is_same<std::size_t, hsize_t>::value) {
152 // `if constexpr` can't be used, thus a reinterpret_cast is needed.
153 data = reinterpret_cast<const hsize_t*>(&(elements._ids[0]));
154 } else {
155 raw_elements.resize(length);
156 std::copy(elements._ids.begin(), elements._ids.end(), raw_elements.begin());
157 data = raw_elements.data();
158 }
159
160 if (H5Sselect_elements(space.getId(), H5S_SELECT_SET, num_elements, data) < 0) {
161 HDF5ErrMapper::ToException<DataSpaceException>("Unable to select elements");
162 }
163
164 return Selection(DataSpace(num_elements), space, details::get_dataset(slice));
165}

168template <typename Derivate>
169template <typename T>
170inline void SliceTraits<Derivate>::read(T& array) const {
171 const auto& slice = static_cast<const Derivate&>(*this);
172 const DataSpace& mem_space = slice.getMemSpace();
173 const details::BufferInfo<T> buffer_info(slice.getDataType(), [slice]() -> std::string {
174 return details::get_dataset(slice).getPath();
175 });
176
177 if (!details::checkDimensions(mem_space, buffer_info.n_dimensions)) {
178 std::ostringstream ss;
179 ss << "Impossible to read DataSet of dimensions " << mem_space.getNumberDimensions()
180 << " into arrays of dimensions " << buffer_info.n_dimensions;
181 throw DataSpaceException(ss.str());
182 }
183 details::data_converter<T> converter(mem_space);
184 read(converter.transform_read(array), buffer_info.data_type);
185 // re-arrange results
186 converter.process_result(array);
187}

189template <typename Derivate>
190template <typename T>
191inline void SliceTraits<Derivate>::read(T* array, const DataType& dtype) const {
192 static_assert(!std::is_const<T>::value,
193 "read() requires a non-const structure to read data into");
194 const auto& slice = static_cast<const Derivate&>(*this);
195 using element_type = typename details::inspector<T>::base_type;
196
197 // Auto-detect mem datatype if not provided
198 const DataType& mem_datatype = dtype.empty() ? create_and_check_datatype<element_type>()
199 : dtype;
200
201 if (H5Dread(details::get_dataset(slice).getId(),
202 mem_datatype.getId(),
203 details::get_memspace_id(slice),
204 slice.getSpace().getId(),
205 H5P_DEFAULT,
206 static_cast<void*>(array)) < 0) {
207 HDF5ErrMapper::ToException<DataSetException>("Error during HDF5 Read: ");
208 }
209}

214inline void SliceTraits<Derivate>::write(const T& buffer) {

216 const DataSpace& mem_space = slice.getMemSpace();
217 const details::BufferInfo<T> buffer_info(slice.getDataType(), [slice]() -> std::string {
218 return details::get_dataset(slice).getPath();
219 });
220
221 if (!details::checkDimensions(mem_space, buffer_info.n_dimensions)) {
222 std::ostringstream ss;
223 ss << "Impossible to write buffer of dimensions " << buffer_info.n_dimensions
224 << " into dataset of dimensions " << mem_space.getNumberDimensions();
225 throw DataSpaceException(ss.str());
226 }
227 details::data_converter<T> converter(mem_space);
228 write_raw(converter.transform_write(buffer), buffer_info.data_type);
229}
