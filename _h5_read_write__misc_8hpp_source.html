35template <typename SrcStrT>
36struct string_type_checker {
37 static DataType getDataType(const DataType&, const DataType&);

40template <>
41struct string_type_checker<void> {
42 inline static DataType getDataType(const DataType& element_type, const DataType& dtype) {
43 // TEMP. CHANGE: Ensure that the character set is properly configured to prevent
44 // converter issues on HDF5 <=v1.12.0 when loading ASCII strings first.
45 // See https://github.com/HDFGroup/hdf5/issues/544 for further information.
46 if (H5Tget_class(element_type.getId()) == H5T_STRING &&
47 H5Tget_cset(dtype.getId()) == H5T_CSET_ASCII) {
48 H5Tset_cset(element_type.getId(), H5T_CSET_ASCII);
49 }
50 return element_type;
51 }
52};

54template <std::size_t FixedLen>
55struct string_type_checker<char[FixedLen]> {
56 inline static DataType getDataType(const DataType& element_type, const DataType& dtype) {
57 DataType return_type = (dtype.isFixedLenStr()) ? AtomicType<char[FixedLen]>()
58 : element_type;
59 // TEMP. CHANGE: See string_type_checker<void> definition
60 if (H5Tget_cset(dtype.getId()) == H5T_CSET_ASCII) {
61 H5Tset_cset(return_type.getId(), H5T_CSET_ASCII);
62 }
63 return return_type;
64 }
65};

67template <>
68struct string_type_checker<char*> {
69 inline static DataType getDataType(const DataType&, const DataType& dtype) {
70 if (dtype.isFixedLenStr())
71 throw DataSetException("Can't output variable-length to fixed-length strings");
72 // TEMP. CHANGE: See string_type_checker<void> definition
73 DataType return_type = AtomicType<std::string>();
74 if (H5Tget_cset(dtype.getId()) == H5T_CSET_ASCII) {
75 H5Tset_cset(return_type.getId(), H5T_CSET_ASCII);
76 }
77 return return_type;
78 }

81template <typename T>
82template <class F>
83BufferInfo<T>::BufferInfo(const DataType& dtype, F getName)
84 : is_fixed_len_string(dtype.isFixedLenStr())
85 // In case we are using Fixed-len strings we need to subtract one dimension
86 , n_dimensions(details::inspector<type_no_const>::recursive_ndim -
87 ((is_fixed_len_string && is_char_array) ? 1 : 0))
88 , data_type(
89 string_type_checker<char_array_t>::getDataType(create_datatype<elem_type>(), dtype)) {
90 if (is_fixed_len_string && std::is_same<elem_type, std::string>::value) {
91 throw DataSetException(
92 "Can't output std::string as fixed-length. "
93 "Use raw arrays or FixedLenStringArray");
94 }
95 // We warn. In case they are really not convertible an exception will rise on read/write
96 if (dtype.getClass() != data_type.getClass()) {
97 std::cerr << "HighFive WARNING \"" << getName()
98 << "\": data and hdf5 dataset have different types: " << data_type.string()
99 << " -> " << dtype.string() << std::endl;
100 }
